## Write BIRD config without kernel export

$ cat /tmp/bird-route-manager/bird-no-export.conf
log stderr all;
router id 172.20.50.1;

protocol device {
  scan time 1;
}

protocol direct direct_service {
  ipv4;
  interface "svc0";
}

protocol static static_lab {
  ipv4;
  route 172.20.99.0/24 blackhole;
}

protocol kernel kernel_ipv4 {
  ipv4 {
    import none;
    export none;
  };
  scan time 1;
}

$ bird -p -c /tmp/bird-route-manager/bird-no-export.conf

$ ip netns exec birdlab bird -c /tmp/bird-route-manager/bird-no-export.conf -s /tmp/bird-route-manager/bird.ctl -P /tmp/bird-route-manager/bird.pid

## Reload BIRD with explicit kernel export

$ cat /tmp/bird-route-manager/bird-export.conf
log stderr all;
router id 172.20.50.1;

protocol device {
  scan time 1;
}

protocol direct direct_service {
  ipv4;
  interface "svc0";
}

protocol static static_lab {
  ipv4;
  route 172.20.99.0/24 blackhole;
}

filter kernel_export_lab {
  if net = 172.20.99.0/24 then accept;
  reject;
}

protocol kernel kernel_ipv4 {
  ipv4 {
    import none;
    export filter kernel_export_lab;
  };
  scan time 1;
}

$ bird -p -c /tmp/bird-route-manager/bird-export.conf

$ cp /tmp/bird-route-manager/bird-export.conf /tmp/bird-route-manager/bird-no-export.conf

$ ip netns exec birdlab birdc -s /tmp/bird-route-manager/bird.ctl configure
