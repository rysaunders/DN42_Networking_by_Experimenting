## Rollback before setup

$ bash -c ip\ netns\ list\ \|\ grep\ -E\ \'\^\(pocket-as2\|pocket-as3\)\(\ \|\$\)\'\ \|\|\ true

## Create two namespaces

$ install -d -m 700 /tmp/wireguard-point-to-point

$ stat -c %a\ %n /tmp/wireguard-point-to-point
700 /tmp/wireguard-point-to-point

$ ip netns add pocket-as2

$ ip netns add pocket-as3

$ ip netns list
pocket-as3
pocket-as2

## Build the underlay

$ ip link add as2-underlay type veth peer name as3-underlay

$ ip link set as2-underlay netns pocket-as2

$ ip link set as3-underlay netns pocket-as3

$ ip -n pocket-as2 addr add 192.0.2.1/30 dev as2-underlay

$ ip -n pocket-as3 addr add 192.0.2.2/30 dev as3-underlay

$ ip -n pocket-as2 link set lo up

$ ip -n pocket-as3 link set lo up

$ ip -n pocket-as2 link set as2-underlay up

$ ip -n pocket-as3 link set as3-underlay up

## Verify the underlay

$ ip -n pocket-as2 route get 192.0.2.2

    cache

## Generate WireGuard keys

$ bash -c umask\ 077\;\ wg\ genkey\ \>\ \'/tmp/wireguard-point-to-point/as2.key\'

$ bash -c wg\ pubkey\ \<\ \'/tmp/wireguard-point-to-point/as2.key\'\ \>\ \'/tmp/wireguard-point-to-point/as2.pub\'

$ bash -c umask\ 077\;\ wg\ genkey\ \>\ \'/tmp/wireguard-point-to-point/as3.key\'

$ bash -c wg\ pubkey\ \<\ \'/tmp/wireguard-point-to-point/as3.key\'\ \>\ \'/tmp/wireguard-point-to-point/as3.pub\'

## Create WireGuard overlay interfaces

$ ip -n pocket-as2 link add wg23 type wireguard

$ ip -n pocket-as3 link add wg23 type wireguard

$ ip -n pocket-as2 addr add 10.42.23.1/30 dev wg23

$ ip -n pocket-as3 addr add 10.42.23.2/30 dev wg23

## Configure WireGuard peers

$ ip netns exec pocket-as2 wg set wg23 private-key /tmp/wireguard-point-to-point/as2.key listen-port 51823 peer 4lgm5stlzBgArTctm6pYmHUJMnY0x0CN9/4DbQozamk= allowed-ips 10.42.23.2/32 endpoint 192.0.2.2:51824

$ ip netns exec pocket-as3 wg set wg23 private-key /tmp/wireguard-point-to-point/as3.key listen-port 51824 peer Q0xaW9G5KDJehRsK4cWHaxig6R6xvwh3qHECoAmWPRM= allowed-ips 10.42.23.1/32 endpoint 192.0.2.1:51823

$ ip -n pocket-as2 link set wg23 up

$ ip -n pocket-as3 link set wg23 up
